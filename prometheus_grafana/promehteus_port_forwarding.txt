# Port-forward Prometheus (9090) from the HPC compute node to localhost.
# Replace HPC_USER and the hostnames with your cluster's login/compute node.
#
# Example command:
(base) user@local ~ % ssh -J $HPC_USER@login.hpc.example.edu \
    -L 9090:localhost:9090 \
    $HPC_USER@compute-node.hpc.example.edu

# You will be prompted for your HPC password (once per hop).
# After connecting, Prometheus will be available at http://localhost:9090
(base) [$HPC_USER@compute-node ~]$ exit
